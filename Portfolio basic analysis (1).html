import numpy as np
import pandas as pd

# Simulated returns and covariance matrix
expected_returns = np.array([0.05, 0.07, 0.12])  # Example expected returns for assets
cov_matrix = np.array([
    [0.01, 0.0018, 0.0011],
    [0.0018, 0.0109, 0.0026],
    [0.0011, 0.0026, 0.0199]
])  # Example covariance matrix
risk_free_rate = 0.03  # Risk-free rate

# Calculate excess returns
excess_returns = expected_returns - risk_free_rate

# Mean-variance optimization
ones = np.ones(len(expected_returns))
inv_cov_matrix = np.linalg.inv(cov_matrix)
weights = np.dot(inv_cov_matrix, excess_returns) / np.dot(ones.T, np.dot(inv_cov_matrix, excess_returns))

# Print results
print("Optimal Portfolio Weights (Mean-Variance Optimization):", weights)

Black-Litterman Model¶
The Black-Litterman model starts with market equilibrium (CAPM-implied returns) and adjusts them based on subjective investor views.


The model combines prior market data (mean returns and covariance) with investor views (expressed as relative or absolute performance expectations) to compute adjusted expected returns. This allows for a more flexible portfolio allocation.


















In [30]:






from numpy.linalg import inv

# Market data
market_weights = np.array([0.4, 0.3, 0.3])  # Market portfolio weights
market_cov = cov_matrix  # Market covariance matrix
market_risk_aversion = 3.0  # Risk aversion coefficient

# Market equilibrium returns
pi = market_risk_aversion * np.dot(market_cov, market_weights)

# Investor's views
P = np.array([[1, 0, -1], [0, 1, -1]])  # Investor's views on relative asset performance
Q = np.array([0.02, 0.01])  # Expected returns of views
omega = np.diag([0.0001, 0.0001])  # Confidence in views

# Black-Litterman posterior returns
M_inverse = inv(inv(market_cov) + np.dot(P.T, inv(omega).dot(P)))
bl_returns = M_inverse.dot(np.dot(inv(market_cov), pi) + np.dot(P.T, inv(omega).dot(Q)))

# Print results
print("Black-Litterman Adjusted Returns:", bl_returns)

CVaR optimization - This is a robust risk measure that's better than traditional variance¶
Purpose: CVaR estimates the expected loss in the worst-case scenario beyond a certain confidence level, offering a more comprehensive risk measure than VaR.
Process:


    

  • Computes losses exceeding the Value at Risk (VaR) threshold.
    • 

  • Useful for risk-averse investors to understand tail risks in the portfolio.
    •

import numpy as np
import cvxpy as cp

def optimize_portfolio_cvar(returns, confidence_level=0.95):
    """
    Optimize portfolio using CVaR minimization.
    :param returns: DataFrame of daily returns.
    :param confidence_level: Confidence level for CVaR.
    :return: Optimal weights for the portfolio.
    """
    n_assets = returns.shape[1]
    weights = cp.Variable(n_assets)  # Portfolio weights
    z = cp.Variable(len(returns))  # Slack variables for CVaR
    VaR = cp.Variable()  # Value at Risk

    # Portfolio return
    portfolio_returns = returns.values @ weights

    # CVaR constraints
    cvar_constraints = [
        z >= -portfolio_returns - VaR,
        z >= 0,
        cp.sum(weights) == 1,  # Weights sum to 1
        weights >= 0  # No shorting
    ]

    # Objective: Minimize CVaR
    cvar = VaR + (1 / ((1 - confidence_level) * len(returns))) * cp.sum(z)
    problem = cp.Problem(cp.Minimize(cvar), cvar_constraints)
    problem.solve()

    return weights.value

optimal_weights = optimize_portfolio_cvar(returns)
print("Optimal Weights:", optimal_weights)

def evaluate_portfolio(returns, weights):
    """
    Evaluate portfolio performance.
    :param returns: DataFrame of daily returns.
    :param weights: Optimized weights.
    :return: Portfolio mean return and standard deviation.
    """
    portfolio_return = np.dot(returns.mean(), weights) * 252  # Annualized return
    portfolio_volatility = np.sqrt(np.dot(weights.T, np.dot(returns.cov() * 252, weights)))  # Annualized volatility
    return portfolio_return, portfolio_volatility

Efficient frontier analysis - Valuable for portfolio optimization visualization¶
The curve's shape shows the tradeoff between risk and return


The upper edge of the curve is the "efficient frontier" - these are the optimal portfolios


Portfolios below the frontier are suboptimal because you could get higher returns for the same risk.


    

  • Portfolio Selection:
    • 



Conservative investors might prefer portfolios on the left (lower risk, lower return)


Aggressive investors might prefer portfolios on the right (higher risk, higher return)


The red dot (max Sharpe ratio) represents the theoretically optimal portfolio for risk-adjusted returns

def plot_efficient_frontier(returns, num_portfolios=10000):
    """
    Plot efficient frontier.
    :param returns: DataFrame of daily returns.
    :param num_portfolios: Number of portfolios to simulate.
    """
    n_assets = returns.shape[1]
    mean_returns = returns.mean() * 252
    cov_matrix = returns.cov() * 252
    results = np.zeros((3, num_portfolios))
    weights_record = []

    for i in range(num_portfolios):
        weights = np.random.random(n_assets)
        weights /= np.sum(weights)
        portfolio_return = np.dot(weights, mean_returns)
        portfolio_volatility = np.sqrt(np.dot(weights.T, np.dot(cov_matrix, weights)))
        results[0, i] = portfolio_volatility
        results[1, i] = portfolio_return
        results[2, i] = results[1, i] / results[0, i]  # Sharpe Ratio
        weights_record.append(weights)

    max_sharpe_idx = np.argmax(results[2])
    max_sharpe_weights = weights_record[max_sharpe_idx]

    plt.figure(figsize=(10, 6))
    plt.scatter(results[0, :], results[1, :], c=results[2, :], cmap='viridis')
    plt.colorbar(label="Sharpe Ratio")
    plt.scatter(results[0, max_sharpe_idx], results[1, max_sharpe_idx], color="red", label="Max Sharpe Ratio")
    plt.title("Efficient Frontier")
    plt.xlabel("Volatility (Risk)")
    plt.ylabel("Return")
    plt.legend()
    plt.show()

plot_efficient_frontier(returns)

from sklearn.decomposition import PCA

def perform_pca(returns):
    """
    Perform PCA on asset returns.
    :param returns: DataFrame of daily returns.
    """
    pca = PCA()
    pca.fit(returns)

def monte_carlo_simulation(returns, weights, num_simulations=1000):
    """
    Perform Monte Carlo simulation.
    :param returns: DataFrame of daily returns.
    :param weights: Portfolio weights.
    :param num_simulations: Number of simulations.
    """
    mean_returns = returns.mean()
    cov_matrix = returns.cov()
    portfolio_returns = []

    for _ in range(num_simulations):
        simulated_returns = np.random.multivariate_normal(mean_returns, cov_matrix, len(returns))
        portfolio_return = np.dot(simulated_returns, weights).mean()
        portfolio_returns.append(portfolio_return)

    plt.figure(figsize=(8, 6))
    plt.hist(portfolio_returns, bins=50, color="skyblue", alpha=0.7)
    plt.title("Monte Carlo Simulated Portfolio Returns")
    plt.xlabel("Portfolio Return")
    plt.ylabel("Frequency")
    plt.show()

monte_carlo_simulation(returns, optimal_weights)
